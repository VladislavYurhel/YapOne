E -> T BODY
   | T BEGIN BODY END ENDEXP
T -> FOR VAR ASSIGN EXP CYCLE CYCLEXP DO
CYCLEXP -> EXP
EXP -> BRACKETS
     | BRACKETS SIGN CONST
     | BRACKETS SIGN EXP
     | BRACKETS SIGN VAR
     | VAR SIGN EXP
     | CONST SIGN EXP
     | VAR
     | CONST
BRACKETS -> OPEN EXP CLOSE
BODY -> INBODY
INBODY -> VAR SIGN EXP ENDEXP BODY
        | VAR SIGN EXP ENDEXP
